                                    function order(words){
                                let arr = words.split(' ');
                                let sortWords = [];
                                let result = '';
                                for (i=0; i<arr.length; i++){
                                  for(j=0; j<arr[j].length; j++){
                                    let num = Number(arr[i][j]);
                                    if(Number.isInteger(num)){
                                      sortWords[num-1] = arr[i];;
                                    }
                                  }
                                }
                                result = sortWords.join(' ');
                                return result;
                              }
